import org.apache.commons.lang3.StringUtils;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.text.SimpleDateFormat;
import java.util.*;

class Person {
    public static void main(String[] args) {

        Person Mike = new Person(ATTRIBUTES.Generous, ATTRIBUTES.Sincere, ATTRIBUTES.Strong,
                ATTRIBUTES.Courageous, ATTRIBUTES.Adventurous, ATTRIBUTES.Ambitious);

        Person Claire = new Person(ATTRIBUTES.Trustworthy, ATTRIBUTES.Caring, ATTRIBUTES.Beautiful,
                ATTRIBUTES.Determined, ATTRIBUTES.Attractive, ATTRIBUTES.Ambitious, ATTRIBUTES.Smart,
                ATTRIBUTES.Nerdy, ATTRIBUTES.GreatSinger, ATTRIBUTES.GreatCook);

        try {
            Mike.wouldLikeToDate(Claire);

        } catch (WhatDoYouSay decision) {
            if (YES.contains(decision.toString().toUpperCase())) {
                DateDate date = new DateDate("24-08-2019 15:00:00");//Aug 24, 2019 @ 10AM PDT
                String[] DateChoices = {"Adventurous Hike", "Fancy Dinner", "Swing Dance",
                        "Bike Ride", "Watch Musical", "Sky Dive", "Rock Climb", "Visit Leavenworth",
                        "Surprise Me"};

                if (date.worksFor(Claire) && StringUtils.isNotBlank(Claire.picksActivity(DateChoices))) {
                    System.out.print(decision.getMessage() + Claire.picksActivity(DateChoices));

                } else {
                    System.out.print("Let's try for another time and/or let's do something not listed");

                }
            } else {
                System.out.print("Mike.heart.break()");
            }
        }
    }

    /**
     * Code to make this work, I had to make sure everything compiled
     */
    private ATTRIBUTES[] personalAttributes;
    private static Random random = new Random();
    private static Set<String> YES = new HashSet<>();

    static {
        String[] PositiveResponses = {"SURE", "YA", "OF COURSE", "HECK YEAH", "YES", "ONE SECOND", "ABSOLUTELY",
                "YEP", "YUP", "YAAAAAS", "TOTES", "YOU BET", "DEFINITELY", "GLADLY", "AYE", "OBVIOUSLY", "OK"};
        YES.addAll(Arrays.asList(PositiveResponses));
    }

    private enum ATTRIBUTES {
        Positive, Sincere, Ambitious, Focused, Trustworthy, Curious, Adventurous, Skilled, Strong,
        Generous, Cooperative, Courageous, Insightful, Patient, Kind, Determined, Beautiful, Caring,
        GreatCook, GreatSinger, Funny, Smart, Attractive, Nerdy
    }

    private Person(ATTRIBUTES... attributes) {
        this.personalAttributes = attributes;
    }

    private static String readLine() {
        String input = "";
        try {
            BufferedReader bufferedReader = new BufferedReader(new InputStreamReader(System.in));
            input = bufferedReader.readLine();
        } catch (IOException e) {
            System.out.print("Failed or interrupted I/O operations. Please try again");
        }
        return input;
    }

    private String picksActivity(String[] DateOptions) {
        String dateActivity = readLine();
        HashSet DateOptionsSet = new HashSet(Arrays.asList(DateOptions));
        while (!DateOptionsSet.contains(dateActivity)) {
            dateActivity = readLine();
        }
        return dateActivity;
    }

    static class WhatDoYouSay extends Exception {
        WhatDoYouSay(String message) {
            super(message);
        }
    }

    static class DateDate extends SimpleDateFormat {
        DateDate(String date) {
            super(date);
        }

        public boolean worksFor(Person person) {
            String dateDecision = readLine().toUpperCase();
            return YES.contains(dateDecision);
        }
    }

    private void wouldLikeToDate(Person person) throws WhatDoYouSay {
        String decision = readLine();
        throw new WhatDoYouSay(decision);
    }
}
